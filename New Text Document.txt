package com.company;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

public class Main {
    public static boolean check(int arr[],int x,int k,int n)
    {
        int i=0;
        for( i=0;i<n;i=i+k)
        {
            int flag=0;
            for(int j=0;j<k;j++)
            {
                if(i+j<n && arr[i+j]==x) {
                    flag = 1;
                    break;
                }


                }
            if(flag==0)
                return false;
            }
        if(i>=n)
            return true;
        else
            return false;
        }

    public static void main(String[] args)throws IOException {

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int n,k,x;
        // for taking input array separated by space we take input as a string in then split it up
        //taking input string of n k x as 12 3 3
        String s[]= br.readLine().split(" ");
        n =Integer.parseInt(s[0]);
        k=Integer.parseInt(s[1]);
        x=Integer.parseInt(s[2]);
        int arr[]=new int[n];
        //array element seperated by comas
        String str[]=br.readLine().split(" ");
        for(int i=0;i<n;i++)
              arr[i]=Integer.parseInt(str[i]);
        if(check(arr,x,k,n))
        {
            System.out.println("YES");
        }
        else
            System.out.println("No");
    }
}
